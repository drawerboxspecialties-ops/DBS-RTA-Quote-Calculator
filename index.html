function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const data = sheet.getDataRange().getValues(); // 1. PO SEARCH (Now Case-Insensitive) if (e.parameter.po) { const searchTarget = e.parameter.po.toString().toLowerCase(); // Convert search to lowercase for (let i = 1; i < data.length; i++) { // Convert sheet data to lowercase for comparison if (data[i][3].toString().toLowerCase() === searchTarget) { return ContentService.createTextOutput(JSON.stringify({status: "found", data: data[i]})) .setMimeType(ContentService.MimeType.JSON); } } } // 2. CUSTOMER SEARCH if (e.parameter.customer) { const nameToFind = e.parameter.customer.toLowerCase(); let matches = []; let seenAccounts = new Set(); for (let i = data.length - 1; i >= 1; i--) { let custName = data[i][1].toString(); let accNum = data[i][2].toString(); if (custName.toLowerCase().includes(nameToFind) && !seenAccounts.has(accNum)) { matches.push({ name: custName, account: accNum, ship: data[i][7], bill: data[i][8] }); seenAccounts.add(accNum); } } return ContentService.createTextOutput(JSON.stringify({status: "list", data: matches})) .setMimeType(ContentService.MimeType.JSON); } // 3. MATERIAL DROPDOWN SEARCH if (e.parameter.getMaterials) { let cabs = new Set(); let doors = new Set(); for (let i = 1; i < data.length; i++) { if(data[i][9]) data[i][9].split("; ").forEach(item => { if(item.includes("(")) cabs.add(item.split("(")[0]); }); if(data[i][10]) data[i][10].split("; ").forEach(item => { if(item.includes("(")) doors.add(item.split("(")[0]); }); } return ContentService.createTextOutput(JSON.stringify({cabs: Array.from(cabs), doors: Array.from(doors)})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({status: "not_found"})).setMimeType(ContentService.MimeType.JSON); } // doPost remains the same to handle Overwriting/Saving function doPost(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheets()[0]; const d = e.parameter; const data = sheet.getDataRange().getValues(); let rowToUpdate = -1; for (let i = 1; i < data.length; i++) { // Also made the Overwrite check case-insensitive just in case if (data[i][3].toString().toLowerCase() === d.PO.toString().toLowerCase()) { rowToUpdate = i + 1; break; } } const rowData = [ new Date(), d.Customer, d.Account, d.PO, d.Date, d.By, d.ShipMethod, d.ShipAddr, d.BillAddr, d.Cabinets, d.Doors, d.MatCost, d.Margin, d.DoorCost, d.Shipping, d.Sheets, d.Labor, d.TaxRate, d.Subtotal, d.TaxAmount, d.GrandTotal, d.Discount ]; if (rowToUpdate > -1) { sheet.getRange(rowToUpdate, 1, 1, rowData.length).setValues([rowData]); } else { sheet.appendRow(rowData); } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService.createTextOutput("Error: " + err).setMimeType(ContentService.MimeType.TEXT); } finally { lock.releaseLock(); } }
